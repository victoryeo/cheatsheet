# install keras and numpy
conda install -c conda-forge keras
conda install numpy

# do not automaticaly activate conda environment
conda config --set auto_activate_base False

# do not change the prompt
conda config --set changeps1 False

conda activate
conda deactivate

# install python spyder
conda install -c anaconda spyder
conda upgrade spyder

# list the environment
conda env list

# install pyTorch
conda install -c pytorch pytorch

# install pandas_datareader
conda install -c anaconda pandas-datareader

# add python2 to the conda environment
conda create -n python python=2.7
conda activate python
conda deactivate
